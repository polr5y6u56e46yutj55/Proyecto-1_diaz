#define PIN_PIR 7
#define PIN_LED 8

void setup() {
  pinMode(PIN_PIR, INPUT);
  pinMode(PIN_LED, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  controlarMovimiento();
}

//  Funcion que prende luz si hay movimiento
void controlarMovimiento() {

  int estadoMovimiento = digitalRead(PIN_PIR);

  if (estadoMovimiento == HIGH) {
    digitalWrite(PIN_LED, HIGH);
    Serial.println("Movimiento detectado - Luz encendida");
  } else {
    digitalWrite(PIN_LED, LOW);
    Serial.println("Sin movimiento - Luz apagada");
  }
}
