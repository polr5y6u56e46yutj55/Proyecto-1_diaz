#define CANT_OUTPUT 3
#define CANT_INPUT  2

int pinesOutput[CANT_OUTPUT] = {2, 3, 4};
int pinesInput[CANT_INPUT]   = {5, 6};

void setup() {
  Serial.begin(9600);

  configurarPines(pinesOutput, CANT_OUTPUT, pinesInput, CANT_INPUT);

  Serial.println("Pines configurados correctamente.");
}

  // CÃ³digo principal
}

//  Funcion que configura pines como OUTPUT o INPUT
void configurarPines(int pinesOut[], int cantOut, int pinesIn[], int cantIn) {

  // Configurar pines OUTPUT
  for(int i = 0; i < cantOut; i++) {
    pinMode(pinesOut[i], OUTPUT);
  }

  // Configurar pines INPUT
  for(int i = 0; i < cantIn; i++) {
    pinMode(pinesIn[i], INPUT);
  }
}
