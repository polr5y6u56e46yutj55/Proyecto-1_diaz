#define CANT 5

int pines[CANT] = {2, 3, 4, 5, 6};

void setup() {

  for(int i = 0; i < CANT; i++) {
    pinMode(pines[i], OUTPUT);
  }

  encenderTodos(pines, CANT);
}

void loop() {
}

//  Funcion que recibe vector de pines y los pone en HIGH
void encenderTodos(int vectorPines[], int cantidad) {

  for(int i = 0; i < cantidad; i++) {
    digitalWrite(vectorPines[i], HIGH);
  }
}
