void setup() {
  Serial.begin(9600);

  int numero = 20;
  int divisor = 5;

  if (esMultiplo(numero, divisor)) {
    Serial.println("Es multiplo.");
  } else {
    Serial.println("No es multiplo.");
  }
}

void loop() {
}

//  Funcion que verifica m√∫ltiplo
bool esMultiplo(int a, int b) {

  if (b == 0) {
    return false;  // Evita division por cero
  }

  if (a % b == 0) {
    return true;
  } else {
    return false;
  }
}
